// match.js $(document).ready(function() { // Initialisation des données let currentMatch = 1; const maxMatches = 3; // Par exemple, Best-of-3 let playerChoices = []; let stageChoices = []; let scores = { player1: 0, player2: 0 }; // Liste des personnages et stages légaux const playerProfileCharacters = [ { name: "Mario" }, { name: "Link" }, { name: "Samus" }, // Ajoutez d'autres personnages ici ]; const legalStages = [ "Dreamland", "Final Destination", "Yoshis Story", // Ajoutez d'autres stages légaux ici ]; // Charger les images des personnages depuis player_profile_characters function loadCharacterImages() { const characterSection = $('#player-selection'); playerProfileCharacters.forEach(character => { const button = $(``); characterSection.append(button); }); } // Charger la liste des stages légaux function loadStages() { const stageSelect = $('#stage-select'); legalStages.forEach(stage => { const option = $(``); stageSelect.append(option); }); } // Gestion de la sélection des personnages $('#player-selection').on('click', '.character-btn', function() { const character = $(this).data('character'); playerChoices[currentMatch - 1] = character; notify(`Joueur a sélectionné ${character}`); }); // Gestion de la sélection des stages $('#stage-select').on('change', function() { const stage = $(this).val(); stageChoices[currentMatch - 1] = stage; notify(`Stage sélectionné : ${stage}`); }); // Valider la sélection et afficher les résultats $('#submit-choice').click(function() { if (!playerChoices[currentMatch - 1] || !stageChoices[currentMatch - 1]) { alert('Veuillez sélectionner un personnage et un stage pour cette manche.'); return; } // Simuler le résultat de la manche (pour cet exemple, on suppose que joueur 1 gagne toujours) scores.player1++; notify(`Manche ${currentMatch} terminée : Joueur 1 remporte avec ${playerChoices[currentMatch - 1]} sur ${stageChoices[currentMatch - 1]}`); // Mettre à jour l'affichage des résultats updateResults(); // Passer à la manche suivante ou terminer le match currentMatch++; if (currentMatch <= maxMatches) { $('#player-selection .character-btn').prop('disabled', false); $('#stage-select').prop('disabled', false); notify(`Aller à la manche ${currentMatch}`); } else { endMatch(); } }); // Mettre à jour l'affichage des résultats function updateResults() { const resultsSection = $('#match-results'); resultsSection.html(`

Résultats du match

Joueur 1 : ${scores.player1} manches remportées

Joueur 2 : ${scores.player2} manches remportées

`); } // Terminer le match function endMatch() { const resultsSection = $('#match-results'); if (scores.player1 > scores.player2) { resultsSection.append('

Joueur 1 a gagné le match !

'); } else { resultsSection.append('

Joueur 2 a gagné le match !

'); } $('#player-selection .character-btn').prop('disabled', true); $('#stage-select').prop('disabled', true); } // Charger les images des personnages et les stages loadCharacterImages(); loadStages(); // Initialiser l'état du match updateResults(); });
